    <script>
        let currentSlide = 0;
        let activePersonality = 'Balanced';
        let turnCount = 0;
        let globalMediaMode = 'image';

        const slidesData = [
            { tag: "SYSTEM BOOTING", icon: "📥", title: "Welcome to My Synex", desc: "A cyber-grade AI workspace. Dual-pane chat + Canvas, personality-driven models, and real-time memory HUD." },
            { tag: "MULTI-ENGINE", icon: "⚡", title: "Switch Models on the Fly", desc: "Claude Sonnet 4.5, GPT-5.2, and Gemini 3 Pro — swap engines mid-conversation from the workspace header." },
            { tag: "CREATOR SUITE", icon: "✨", title: "Generate Images & Video", desc: "Prompt Nano Banana for images and Sora 2 for video posters — all rendered inline into your Canvas." },
            { tag: "CONTEXT LOADED", icon: "❓", title: "Memory Engine Active", desc: "Track what the AI remembers in real time. Focus Mode hides distractions when it's time to build." }
        ];

        function nextOnboardingSlide() {
            currentSlide++;
            if (currentSlide >= slidesData.length) {
                closeOnboarding();
                return;
            }
            const data = slidesData[currentSlide];
            document.getElementById('ob-badge').innerText = data.tag;
            document.getElementById('ob-icon-wrapper').innerText = data.icon;
            document.getElementById('ob-title').innerText = data.title;
            document.getElementById('ob-desc').innerText = data.desc;

            for (let i = 1; i <= 4; i++) {
                document.getElementById(`dot-${i}`).className = i === (currentSlide + 1) ? "w-5 h-1 rounded-full bg-[#0ea5e9] transition-all" : "w-1.5 h-1 rounded-full bg-gray-800 transition-all";
            }
            if (currentSlide === slidesData.length - 1) {
                document.getElementById('ob-next-btn').innerText = "Finish";
            }
        }

        function closeOnboarding() { document.getElementById('onboarding-container').style.display = 'none'; }
        function restartOnboarding() { currentSlide = -1; document.getElementById('onboarding-container').style.display = 'flex'; document.getElementById('ob-next-btn').innerText = "Continue"; nextOnboardingSlide(); }
        function toggleEngineMenu() { document.getElementById('engine-dropdown').classList.toggle('hidden'); }

        function selectEngine(name) {
            document.getElementById('active-engine-text').innerText = name;
            document.getElementById('hud-engine').innerText = name;
            document.getElementById('chk-claude').classList.toggle('hidden', name !== 'Claude Sonnet 4.5');
            document.getElementById('chk-gpt').classList.toggle('hidden', name !== 'GPT-5.2');
            document.getElementById('chk-gemini').classList.toggle('hidden', name !== 'Gemini 3 Pro');
            toggleEngineMenu();
        }
        function goToTab(target) {
            ['workspace', 'create', 'memory', 'profile', 'settings'].forEach(tab => {
                document.getElementById(`view-${tab}`).classList.toggle('hidden', tab !== target);
                document.getElementById(`nav-item-${tab}`).className = tab === target ? 'flex flex-col items-center gap-1 text-cyan-400 py-1 flex-1 text-center font-bold' : 'flex flex-col items-center gap-1 text-gray-600 py-1 flex-1 text-center';
            });
        }

        function injectQuickPrompt(text) {
            document.getElementById('chat-input').value = text;
            sendChatMessage();
        }

        function sendChatMessage() {
            const input = document.getElementById('chat-input');
            const scroller = document.getElementById('chat-scroller');
            const text = input.value.trim();
            if (!text) return;

            document.getElementById('empty-state').style.display = 'none';
            turnCount++;
            document.getElementById('hud-turns').innerText = turnCount;
            document.getElementById('app-subheader').children[2].innerText = `${turnCount} turns`;

            scroller.innerHTML += `<div class="self-end chat-bubble-user text-xs px-3 py-2.5 rounded-2xl max-w-[85%]">${text}</div>`;
            input.value = '';
            scroller.scrollTop = scroller.scrollHeight;

            setTimeout(() => {
                scroller.innerHTML += `<div class="self-start chat-bubble-ai text-xs px-3 py-2.5 rounded-2xl max-w-[85%]">Canvas Ready. Awaiting your command, Operator.</div>`;
                scroller.scrollTop = scroller.scrollHeight;
            }, 750);
        }

        function setMediaMode(mode) {
            globalMediaMode = mode;
            document.getElementById('tab-media-img').className = mode === 'image' ? 'flex-1 py-2 rounded-lg bg-[#16181d] text-white transition-all' : 'flex-1 py-2 rounded-lg text-gray-500 transition-all';
            document.getElementById('tab-media-vid').className = mode === 'video' ? 'flex-1 py-2 rounded-lg bg-[#16181d] text-white transition-all' : 'flex-1 py-2 rounded-lg text-gray-500 transition-all';
            document.getElementById('prompt-input-area').placeholder = mode === 'image' ? "A cyberpunk cat DJing in Tokyo, neon rain, 4k" : "A rocket taking off from Mars orbit, cinematic wide angle";
            document.getElementById('creator-badge').innerText = mode === 'image' ? "• Nano Banana" : "• Sora 2";
        }

        function triggerRender() {
            const prompt = document.getElementById('prompt-input-area').value.trim();
            const canvas = document.getElementById('render-canvas');
            if(!prompt) return;

            canvas.innerHTML = `<div class="animate-spin rounded-full h-5 w-5 border border-cyan-400 border-t-transparent"></div>`;
            
            setTimeout(() => {
                if(globalMediaMode === 'image') {
                    canvas.innerHTML = `<div class="w-full relative"><img src="https://image.pollinations.ai/p/${encodeURIComponent(prompt)}?width=600&height=400&nologo=true" class="w-full rounded-xl object-cover h-44 border border-dark"><button onclick="triggerRender()" class="absolute bottom-2 right-2 bg-black/70 border border-dark text-[10px] px-2 py-1 rounded-md text-gray-300">🔄 Reroll</button></div>`;
                } else {
                    canvas.innerHTML = `<div class="w-full"><video src="https://assets.mixkit.co/videos/preview/mixkit-abstract-laser-lights-background-41712-large.mp4" controls autoplay loop muted class="w-full rounded-xl object-cover h-44 border border-dark"></video></div>`;
                }
            }, 1000);
        }

        function setPersonality(type) {
            ['Balanced', 'Technical', 'Creative', 'Concise'].forEach(p => {
                document.getElementById(`p-btn-${p}`).className = p === type ? "p-2.5 rounded-xl bg-cyan-400 text-black border border-transparent transition" : "p-2.5 rounded-xl bg-dark-panel text-gray-400 border border-dark transition";
            });
            document.getElementById('hud-personality').innerText = type;
        }

        function toggleFocusMode(isChecked) {
            document.getElementById('app-header').style.display = isChecked ? 'none' : 'flex';
            document.getElementById('app-subheader').style.display = isChecked ? 'none' : 'flex';
        }

        document.getElementById('prompt-input-area').addEventListener('input', function() {
            document.getElementById('char-counter').innerText = `${this.value.length}/500`;
        });
    </script>
