            }
            const innerContainer = contextBox.querySelector('div') || contextBox;
            
            // Build temporary item
            const tagChipText = input.value.split(' ')[0] || "Query";
            const newChip = document.createElement('span');
            newChip.className = "text-[10px] font-mono font-medium px-2 py-0.5 bg-cyan-950/50 text-cyan-400 border border-cyan-900/60 rounded";
            newChip.textContent = `${tagChipText.substring(0,8).toUpperCase()}_NODE`;
            innerContainer.appendChild(newChip);

            // Wipe standard input container

            // Subtle system flash confirmation indicator highlight alerts context response mappings

            activeRibbonTurnText.className = "flex-shrink-0 text-[10px] bg-cyan-900/40 text-cyan-300 border border-cyan-800 rounded-full px-2.5 py-1 font-bold animate-pulse";
